  }

  }

    Message.warn('Example:warn:text')

  }

      </Button>
      <Button onClick={this.onClick4} theme='yellow'>
        Message.warn
      </Button>
      <Button onClick={this.onClick2}>
        Message.error
      </Button>
      <style>
        {`
          .message-box {
            padding: 5px;
          }
          .message-box button {
              margin: 10px;
            }
        `}
      </style>
    </div>
    )
  }
}
